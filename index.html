Import Example
import VoiceKernel from './voice-kernel.js'

const mic = await VoiceKernel.create({
    mode: 'passthrough',
    gain: 1.0,
    chunkDuration: 250,
    onChunk: (samples) => {
        // feed to STT, save to file, etc.
        console.log(samples)
    }
})

mic.setGain(0.5)
mic.setMode('both')
mic.destroy()
